Maurer Rose

Here's a Python script that generates a Maurer Rose:


import numpy as np
import matplotlib.pyplot as plt
from itertools import cycle

def get_rose_xy(n, d):
    k = d * np.linspace(0, 361, 361)
    r = np.sin(np.radians(n * k))
    x = r * np.cos(np.radians(k))
    y = r * np.sin(np.radians(k))
    return x, y

def draw_rose(ax, n, d, colors):
    x, y = get_rose_xy(n, d)
    color_cycle = cycle(colors)
    
    for i in range(1, len(x)):
        ax.plot([x[i-1], x[i]], [y[i-1], y[i]], c=next(color_cycle), lw=0.5)
    
    ax.axis('equal')
    ax.axis('off')

n = 6
d = 71
colors = ['tab:blue', 'tab:orange', 'tab:green', 'tab:red']
fig, ax = plt.subplots()
draw_rose(ax, n, d, colors)
plt.show()
